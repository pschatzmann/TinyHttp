206 bool onRequest(const char* path) {
207 HttpLogger.log(Info,"onRequest %s", path);
208
209 bool result = false;
210 // check in registered handlers
211 Str pathStr{path};
212 // if path contains a query string, remove it
213 int pos = pathStr.indexOf("?");
214 if (pos >= 0){
215 pathStr.substring(pathStr, 0, pos);
216 HttpLogger.log(Debug, "onRequest - pathStr: %s", pathStr.c_str());
217 }
218
219 for (auto it = handler_collection.begin() ; it != handler_collection.end(); ++it) {
220 HttpRequestHandlerLine *handler_line_ptr = *it;
221 HttpLogger.log(Info,"onRequest - checking: %s %s %s", nullstr(handler_line_ptr->path), methods[handler_line_ptr->method], nullstr(handler_line_ptr->mime));
222
223 if (pathStr.matches(handler_line_ptr->path)
224 && request_header.method() == handler_line_ptr->method
225 && matchesMime(handler_line_ptr->mime, request_header.accept())) {
226 // call registed handler function
227 HttpLogger.log(Info,"onRequest %s","->found");
228 handler_line_ptr->fn(this, path, handler_line_ptr);
229 result = true;
230 break;
231 }
232 }
233 return result;
234 }
