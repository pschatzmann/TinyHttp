740 static char* itoa(int n, char s[]) {
741 int i, sign;
742 if ((sign = n) < 0) /* record sign */
743 n = -n; /* make n positive */
744 i = 0;
745 do { /* generate digits in reverse order */
746 s[i++] = n % 10 + '0'; /* get next digit */
747 } while ((n /= 10) > 0); /* delete it */
748 if (sign < 0) s[i++] = '-';
749 s[i] = '\0';
750 reverse(s);
751 return s;

791 unsigned long frac;

799 frac = (int(val) - val) * mult;
800 unsigned long frac1 = frac;
